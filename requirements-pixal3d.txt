# Pixal3Dジェネレータ用の隔離環境 (.venv-pixal3d) の依存関係
#
# 重要: 既存の .venv には絶対にインストールしないこと。Pixal3D は Python 3.10 +
# 独自の依存ピン(trimesh==4.10.1 等)を要求するため、専用venvを分離する:
#
#   uv venv .venv-pixal3d --python 3.10
#
# --- 1. torch (Blackwell sm_120 対応 = cu128ビルド) --------------------------
# 実機検証済み: torch==2.11.0+cu128 / torchvision==0.26.0+cu128
#
#   uv pip install --python .venv-pixal3d/bin/python torch torchvision \
#       --index-url https://download.pytorch.org/whl/cu128
#
# --- 2. このファイルのpip依存 -------------------------------------------------
#
#   uv pip install --python .venv-pixal3d/bin/python -r requirements-pixal3d.txt
#
# --- 3. CUDA拡張のソースビルド (nvcc必須) -------------------------------------
# システムnvcc既定はCUDA 13.0だが、torchのcu128と一致するCUDA 12.8を明示すること。
# CUDACXX の明示が重要(cmakeが /usr/bin/nvcc = 13.0 を拾い、新しいglibcヘッダで
# ビルドが失敗するため):
#
#   export CUDA_HOME=/usr/local/cuda-12.8
#   export CUDACXX=/usr/local/cuda-12.8/bin/nvcc
#   export PATH="$CUDA_HOME/bin:$PATH"
#   export TORCH_CUDA_ARCH_LIST="12.0"
#
#   # o-voxel (TRELLIS.2同梱。CuMesh/FlexGEMMもgitから自動ビルドされる)
#   git clone -b main --recursive https://github.com/microsoft/TRELLIS.2.git third_party/TRELLIS.2
#   uv pip install --python .venv-pixal3d/bin/python --no-build-isolation third_party/TRELLIS.2/o-voxel
#
#   # nvdiffrast (o_voxel.postprocess.to_glb が必須依存。PyPI配布なし)
#   uv pip install --python .venv-pixal3d/bin/python --no-build-isolation \
#       "git+https://github.com/NVlabs/nvdiffrast.git@v0.4.0"
#
#   # NATTEN (NAF特徴アップサンプラが実行時にtorch.hub経由で使用。sm_120ビルド実測9分強)
#   NATTEN_CUDA_ARCH="12.0" NATTEN_N_WORKERS=8 uv pip install \
#       --python .venv-pixal3d/bin/python natten==0.21.0 --no-build-isolation
#
# --- 4. Pixal3D本体 -----------------------------------------------------------
#   git clone https://github.com/TencentARC/Pixal3D third_party/Pixal3D
#   (pipインストール不要。server/generators/pixal3d.py が sys.path 経由でimportする)
#
# 注意:
# - flash_attn は導入しない。ATTN_BACKEND=sdpa / SPARSE_ATTN_BACKEND=sdpa で
#   PyTorch SDPA バックエンドを使用する(Blackwellでflash_attnのプリビルドが
#   torch ABI不一致のため。launch.json の image3d-server-pixal3d 参照)。
# - MoGe(カメラFOV自動推定)は導入しない。固定FOV(IMAGE3D_PIXAL3D_FOV)を使う。
# - Pixal3D公式のrembgモデル briaai/RMBG-2.0 はHFゲート付きのためロードせず、
#   本アプリ側の背景除去(rembg CPU / server/preprocess.py)の結果(RGBA)を渡す。
# - utils3d はレンダリング/学習系のみで使用されるため導入不要。
# - NAF重み(~2.5MB)は初回実行時に torch.hub が valeoai/NAF から自動DLする。
# - モデル重み: TencentARC/Pixal3D 約23GB + DINOv3 (camenduru/dinov3-vitl16) 約1.2GB
#   (初回実行時にHuggingFaceから自動DL)。
